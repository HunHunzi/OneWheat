# 初识虚拟 DOM

  • 虚拟 DOM 的性能：声明式的更新性能消耗 = 找出差异的性能的消耗 + 应用差异的性能消耗
  • 而真实 DOM 需要重新渲染所有的 DOM 节点
  • 其实虚拟 DOM 就是一个 JS 对象，里面包含着 DOM 的结构和属性，通过这个对象来描述真实的 DOM 结构，通过渲染器将虚拟 DOM 渲染成真实的 DOM
//一个简单的虚拟 DOM
const vnode = {
  tag: "div", //标签
  props: {
    //属性
    className: "container",
    id: "app",
  },
  children: [
    //子节点
    {
      tag: "h1",
      props: null,
      children: ["hello"],
    },
    {
      tag: "p",
      props: null,
      children: ["world"],
    },
  ],
};

// 写一个简单的渲染器
function renderer(vnode, container) {
  //使用 vnode.tag 作为标签创建元素
  const el = document.createElement(vnode.tag);
  //遍历 vnode.data 设置属性,事件
  for (const key in vnode.props) {
    //如果 key 是 on 开头的,说明是事件
    if (/^on/.test(key)) {
      el.addEventListener(key.slice(2).toLowerCase(), vnode.props[key]); //绑定事件
    }
    //如果 children 是 字符串，说明是文本
    if (typeof vnode.props[key] === "string") {
      el.appendChild(document.createTextNode(vnode.props[key])); //设置文本
    } else if (Array.isArray(vnode.children)) {
      // 递归调用 render 函数渲染子节点，使用当前 el 元素作为挂载点
      vnode.children.forEach((vnode) => renderer(vnode, el));
    }
    container.appendChild(el); //将 el 挂载到 container 上
  }
}

//然后我们调用渲染器
renderer(vnode, document.getElementById("app"));
// 渲染器的精髓在于它会精确地找到修改后的节点，而无需重新渲染整个 DOM 树

# 组件的本质

  • 组件就是一组 DOM 元素的封装
//一个简单的组件
const MyComponent = function () {
  return {
    tag: "div",
    props: {
      onClick: () => {
        alert("click");
      },
    },
    children: "click me",
  };
};
//组件的返回值也是虚拟 DOM
const vnode = {
  tag: MyComponent,
};

对于组件的渲染器，我们只需要简单地修改一下，typeof vnode.tag === 'function' 时，我们就调用一下组件函数，然后再将返回的 vnode 传入渲染器即可,但是需要对组件的挂载函数做一点小小的修改

//组件挂载
function mountComponent(vnode, container) {
  // 调用组件函数，得到组件对应的虚拟 DOM 对象
  const subTree = vnode.tag();
  // 递归调用 render 函数进行渲染
  renderer(subTree, container);
}

当然我们可以举一反三，使用一个数组来代表组件，只要我们简单修改一下渲染器的判断条件即可

# 模板的工作原理

  • 由于 vue 使用的是声明式语言，所以需要使用编译器，将模板编译成虚拟 DOM，然后再渲染到页面上，<template></template>标签就是编译器的入口，编译器会将模板编译成 render 函数，然后再调用 render 函数生成虚拟 DOM，最后再渲染到页面上。
  • 对于渲编译器来说，想要提高它的效率，那需要用到一些小技巧，比如说缓存，缓存就是将编译好的 render 函数缓存起来，下次再编译的时候，如果模板没有发生变化，那么就直接使用缓存的 render 函数，而不需要重新编译，这样就可以提高编译器的效率。或者是对于一些静态节点，我们可以直接将它编译成字符串，然后再渲染的时候，直接将字符串插入到 DOM 中，这样就可以减少一次渲染的过程，提高渲染的效率。

# 响应式系统的基本原理

# 副作用

副作用函数是指在函数执行过程中，会对外部产生影响的函数，比如说修改全局变量，，而响应式系统的核心就是通过副作用函数来实现的，当数据发生变化的时候，会触发副作用函数，从而实现数据的更新。

//一个简单的副作用函数
const obj = { text: "hello" };
function effect() {
  document.body.innerText = obj.text; //effect函数会修改全局变量
}
effect(); // 我们希望当值变化后，effect函数会重新执行，那么obj.text就是一个响应式数据

# 响应式数据的基本实现

如何才能让数据变成响应式数据？有如下两条线索

  • 但副作用函数 effect 执行的时候，会读取全局变量 obj.text
  • 当 obj.text 发生变化的时候，我们希望 effect 函数会重新执行 所以我们可以设置一个桶，当 obj.text 发生变化的时候，我们就将 effect 函数放入桶中，当 obj.text 发生变化的时候，我们就遍历桶中的函数，然后执行它们，这样就可以实现数据的更新了。 如此一个响应式系统的工作流程大概如下：
  1. 当我们执行 effect 函数的时候，会将 effect 函数放入桶中
  2. 当 obj.text 发生变化的时候，会遍历桶中的函数，然后执行它们
//一个简单的响应式系统
//桶
const bucket = new WeakMap(); //使用 WeakMap 作为桶，因为 WeakMap 的 key 是弱引用，当 key 对象被回收的时候，value 也会被回收

const obj = new Proxy(data, {
  //拦截读取
  get(target, key) {
    // 如果没有activeEffect，说明当前的 key 不是响应式数据，直接返回
    if (!activeEffect) return target[key];
    // 将当前的 key 和 activeEffect(effect函数) 对应起来.key-->effect
    let depsMap = bucket.get(target);
    if (!depsMap) bucket.set(target, (depsMap = new Map()));
    //根据 key 从 depsMap 中获取deps,它是一个 Set 集合，里面存放着 effect 函数，里面存放着所有与key相关的effect函数
    let deps = depsMap.get(key);
    if (!deps) depsMap.set(key, (deps = new Set())); //如果 deps 不存在，那么就创建一个
    deps.add(activeEffect); //将当前的 effect 函数放入 deps 中
    //返回属性值
    return target[key];
  },
  //拦截修改
  set(target, key, value) {
    //修改属性值
    target[key] = value;
    //获取 key 对应的 deps
    const depsMap = bucket.get(target);
    if (depsMap) {
      const deps = depsMap.get(key);
      if (deps) {
        deps.forEach((effect) => effect()); //遍历 deps 中的 effect 函数，然后执行它们
      }
    }
    return true;
  },
  //可将拦截函数单独封装进一个函数中
});

为什么使用 WeakMap 作为桶？

  • WeakMap 的数据结构，WeakMap 的键的样式对象是 target，它的值是一个 Map 实例，Map 实例的键是 target 的属性名，值是一个 Set 实例，Set 实例的值是 effect 函数，这样就形成了一个桶的结构，当 target 被回收的时候，它对应的 Map 实例也会被回收，这样就不会造成内存泄漏了。
  • 因为 WeakMap 的 key 是弱引用，当 key 对象被回收的时候，value 也会被回收，这样就不会造成内存泄漏了。
  • target 事件属性可返回事件的目标节点（触发该事件的节点），如生成事件的元素、文档或窗口。

# 分支切换与 cleanup

  1. 分支切换的定义：
const data ={ok:true,text:'hello'}
const obj = new Proxy(data,{/* */})

effect(function effectFn){
  document.body.innerText = obj.ok?obj.text:'nothing'
}

在 effectFn 中存在一个三元表达式，它会根据 obj.ok 的值进行切换不同的代码分支，这就是分支切换 其内部的对应关系如下：

//分支切换的对应关系
const branchesMap = new Map([
  [true, () => (document.body.innerText = obj.text)],
  [false, () => (document.body.innerText = "nothing")],
]);

我们希望当 obj.ok 变化使，只有对应的分支才会执行，而不是所有的分支都会执行，但是由于 key-->effect-->(ok,text)，遗留的副作用函数会导致不必要的更新，所以我们需要对其进行优化

2.解决方法（cleanup） 解决的思路就是每次执行 effectFn 的时候，都会将上一次的 effectFn 从桶中删除，然后再将当前的 effectFn 放入桶中，这样就可以保证每次只有对应的分支才会执行，而不是所有的分支都会执行，这使得每一个 effectFn 都是会对应相对的以来集合，这样就可以实现精确更新了

//  用一个全局变脸存储被注册的effect函数
let activeEffect = null
function effect(effectFn){
  const effectFn =()=>{
    // 当effectFn执行的时候，设置为激活
    activeEffect = effectFn
    effectFn()
  }
  // activeEffect.deps用来存储所有与该effectFn相关的deps
  effectFn.deps = []
  // 执行effectFn
  effectFn()
}
// 在get中将effectFn放入deps中
//将get单独封装成track

function (target,key)
{
  // 没有activeEffect，说明当前的key不是响应式数据，直接返回
  if(!activeEffect) return target[key]
  // 将当前的key和activeEffect(effect函数)对应起来
  let depsMap = bucket.get(target)
  if(!depsMap) bucket.set(target,depsMap = new Map())
  // 根据key从depsMap中获取deps,它是一个Set集合，里面存放着effect函数，里面存放着所有与key相关的effect函数
  let deps = depsMap.get(key)
  if(!deps) depsMap.set(key,deps = new Set()) //如果deps不存在，那么就创建一个
  // 将当前的effect函数放入deps中
  deps.add(activeEffect)
  //deps就是一个与当前副作用函数存在联系的依赖集合
  activeEffect.deps.push(deps)
}

有了这个联系后我们就可以在每次副作用函数执行时，根据 effectFn.deps 所有相关联的依赖集合，继而从副作用函数从依赖函数中删除

function cleanup(effectFn) {
  // 获取effectFn对应的deps
  const deps = effectFn.deps;
  // 遍历deps
  for (let dep of deps) {
    // 从dep中删除effectFn
    dep.delete(effectFn);
  }
}
//cleanup函数接收副作用函数作为参数，然后遍历effectFn.deps，然后从deps中删除effectFn，最后重置effectFn.deps
//在track函数中其实是存在死循环的，因为在track函数中会调用effectFn，而effectFn中又会调用track函数，所以我们需要在effectFn中将track函数从effectFn.deps中删除。而在语言规范中forEach遍历set集合时，如果改制被删除又被重新添加，则该遍历不会结束

function trigger(target, key) {
  // 获取target中key对应的deps
  const depsMap = bucket.get(target);
  if (!depsMap) return;
  const effects = depsMap.get(key);

  // 遍历effects
  const effectionsToRun = new Set(effects); //使用set集合来存储effects，因为set集合中不会有重复的effect函数
  effectsToRun.forEach((effectFn) => effectFn());
}
  • 总而言之对于分支切换，我们需要将 effectFn 从桶中删除，然后再将当前的 effectFn 放入桶中，这样就可以保证每次只有对应的分支才会执行，而不是所有的分支都会执行，这使得每一个 effectFn 都是会对应相对的以来集合，这样就可以实现精确更新了

# 嵌套的 effect 和 effect

effect(()=>{
  document.body.innerText = obj.text
  effect(()=>{
    document.body.innerText = obj.text
  })
})
// 例如
effect((=>{
  Foo.render()
  effect(()=>{
    Bar.render()
  })
}))

我们希望的结构是:

  • data
    • foo -effectFn1
    • bar -effectFn2

理想状态下我们修改执行 foo 时会触发 Fn1,和嵌套在里面的 Fn2，修改 bar 时会触发 Fn2，但实际上我们修改 foo 时，它会出初始 Fn1,Fn2,然后直接执行 Fn2,只是因为我们在存储时实际上时单层的，Fn2 会覆盖原本的 Fn1，所以不会触发先写入的 Fn1。解决方法是使用栈。

let activeEffect = null;
const effectStack = []; //栈

function effect(fn) {
  const effectFn = () => {
    cleanup(effectFn); //在执行effectFn之前，先将effectFn从effectStack中删除
    // 当effectFn执行的时候，设置为激活
    activeEffect = effectFn;
    effectStack.push(effectFn); //将effectFn放入栈中
    fn(); //执行fn
    //执行完fn后，将effectFn从栈中删除,并且将activeEffect还原成当前的值（栈顶的值）
    effectStack.pop();
    activeEffect = effectStack[effectStack.length - 1];
  };
  // activeEffect.deps用来存储所有与该effectFn相关的deps
  effectFn.deps = [];
  // 执行effectFn
  effectFn();
}

# 避免无限循环

const obj = { foo: "foo" };
effect(() => {
  obj.foo = obj.foo + "foo";
});
